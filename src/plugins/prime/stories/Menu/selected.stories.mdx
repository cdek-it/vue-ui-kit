#### –•—É–∫ `useMenu`

```ts
import { ref } from 'vue';
import type { MenuItem } from 'primevue/menuitem';
import { selectedMenuItemClass } from '@cdek-it/vue-ui-kit';

export const useMenu = () => {
  const items = ref<MenuItem[]>([
    {
      label: 'File',
      items: [
        { label: 'New' },
        { label: 'Open' },
        {
          label: 'Export',
          items: [
            { label: 'PDF' },
            { label: 'Excel' },
          ],
        },
      ],
    },
    {
      label: 'Edit',
      items: [
        { label: 'Undo' },
        { label: 'Redo' },
      ],
    },
  ]);

  const clearselectedMenuItemClass = (menuItems: MenuItem[]) => {
    menuItems.forEach((item) => {
      if (item.class?.includes(selectedMenuItemClass)) {
        item.class = item.class
          .split(' ')
          .filter((c: string) => c !== selectedMenuItemClass)
          .join(' ')
          .trim() || undefined;
      }
      if (item.items) {
        clearselectedMenuItemClass(item.items as MenuItem[]);
      }
    });
  };

  const onClickItem = (item: MenuItem) => {
    clearselectedMenuItemClass(items.value);
    item.class = item.class ? `${item.class} ${selectedMenuItemClass}` : selectedMenuItemClass;
  };

  const addCommandHandler = (menuItems: MenuItem[]) => {
    menuItems.forEach((item) => {
      item.command = () => onClickItem(item);
      if (item.items) {
        addCommandHandler(item.items as MenuItem[]);
      }
    });
  };

  addCommandHandler(items.value);

  return { items };
};
```
